    <script>

        function decode() {

            const cipherText = document.getElementById("cipherText").value.toUpperCase();

            const shift = parseInt(document.getElementById("shift").value);

            let decodedText = "";


            for (let i = 0; i < cipherText.length; i++) {

                const char = cipherText[i];

                if (char >= 'A' && char <= 'Z') {

                    const decodedChar = String.fromCharCode(((char.charCodeAt(0) - 65 - shift + 26) % 26) + 65);

                    decodedText += decodedChar;

                } else {

                    decodedText += char; // Sonderzeichen und Leerzeichen bleiben gleich

                }

            }


            document.getElementById("decodedText").innerText = `Entschlüsselter Text: ${decodedText}`;

        }
